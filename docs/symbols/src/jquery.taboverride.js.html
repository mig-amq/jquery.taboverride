 29 // Use AMD if available
 30 (function ( factory ) {
 31     "use strict";
 32 
 33     if ( typeof define === "function" && define.amd ) {
 34         // AMD - Register as an anonymous module
 35         // Files must be concatenated using an AMD-aware tool such as r.js
 36         define( [ "jquery", "taboverride" ], factory );
 37     } else {
 38         // No AMD - Use global variables instead
 39         factory( jQuery, TABOVERRIDE );
 40     }
 41 }(function ( $, TABOVERRIDE ) {
 42     "use strict";
 43 
 44     /**
 45      * the tabOverride method - Tabs will be overridden if enable is true.
 46      *
 47      * @param  {Boolean} [enable=true]  whether Tab Override should be enabled
 48      *                                  for the element(s)
 49      * @param  {String}  [selector]     the selector string for delegated events
 50      * @return {Object}                 the jQuery object
 51      *
 52      * @name tabOverride
 53      * @memberOf jQuery.fn
 54      * @function
 55      * @namespace
 56      */
 57     var tabOverride = $.fn.tabOverride = function ( enable, selector ) {
 58 
 59         var enablePlugin = !arguments.length || enable,
 60             isDelegated = typeof selector === "string";
 61 
 62         if ( isDelegated ) {
 63             // Always remove the event handlers
 64             this.off( ".tabOverride", selector );
 65 
 66             if ( enablePlugin ) {
 67                 this
 68                     .on( "keydown.tabOverride", selector, TABOVERRIDE.overrideKeyDown )
 69                     .on( "keypress.tabOverride", selector, TABOVERRIDE.overrideKeyPress );
 70             }
 71         } else {
 72             // The jQuery object acts as an array of elements, so it can be passed
 73             // to TABOVERRIDE.set()
 74             TABOVERRIDE.set( this, enablePlugin );
 75         }
 76 
 77         // Return the jQuery object
 78         return this;
 79     };
 80 
 81     /**
 82      * Gets or sets the tab size for all elements that have Tab Override enabled.
 83      * 0 represents the tab character.
 84      *
 85      * @param  {Number}          [size]  the tab size
 86      * @return {Number|Function}         the tab size or the tabOverride function
 87      *
 88      * @name tabSize
 89      * @function
 90      * @memberOf jQuery.fn.tabOverride
 91      */
 92     tabOverride.tabSize = TABOVERRIDE.tabSize;
 93 
 94     /**
 95      * Gets or sets the auto indent setting. True if each line should be
 96      * automatically indented (default = false).
 97      *
 98      * @param  {Boolean}          [enable]  whether auto indent should be enabled
 99      * @return {Boolean|Function}           whether auto indent is enabled or the
100      *                                      tabOverride function
101      *
102      * @name autoIndent
103      * @function
104      * @memberOf jQuery.fn.tabOverride
105      */
106     tabOverride.autoIndent = TABOVERRIDE.autoIndent;
107 
108     /**
109      * Gets or sets the tab key combination.
110      *
111      * @param  {Number}        keyCode             the key code of the key to use for tab
112      * @param  {String[]}      [modifierKeyNames]  the modifier key names - valid names are
113      *                                             'alt', 'ctrl', 'meta', and 'shift'
114      * @return {String|Function}                   the current tab key combination or the
115      *                                             tabOverride function
116      *
117      * @name tabKey
118      * @function
119      * @memberOf jQuery.fn.tabOverride
120      */
121     tabOverride.tabKey = TABOVERRIDE.tabKey;
122 
123     /**
124      * Gets or sets the untab key combination.
125      *
126      * @param  {Number}        keyCode             the key code of the key to use for untab
127      * @param  {String[]}      [modifierKeyNames]  the modifier key names - valid names are
128      *                                             'alt', 'ctrl', 'meta', and 'shift'
129      * @return {String|Function}                   the current untab key combination or the
130      *                                             tabOverride function
131      *
132      * @name untabKey
133      * @function
134      * @memberOf jQuery.fn.tabOverride
135      */
136     tabOverride.untabKey = TABOVERRIDE.untabKey;
