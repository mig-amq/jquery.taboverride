 29 // Use AMD if available
 30 (function ( factory ) {
 31     "use strict";
 32 
 33     if ( typeof define === "function" && define.amd ) {
 34         // AMD - Register as a named module
 35         // Using a named module allows for non-AMD-aware concatenation
 36         define( "jquery.taboverride", [ "jquery", "taboverride" ], factory );
 37     } else {
 38         // No AMD - Use global variables instead
 39         factory( jQuery, TABOVERRIDE );
 40     }
 41 }(function ( $, TABOVERRIDE ) {
 42     "use strict";
 43 
 44     /**
 45      * the tabOverride method - Tabs will be overridden if enable is true.
 46      *
 47      * @param  {Boolean} [enable=true]  whether Tab Override should be enabled
 48      *                                  for the element(s)
 49      * @return {Object}                 the jQuery object
 50      *
 51      * @name tabOverride
 52      * @memberOf jQuery.fn
 53      * @function
 54      * @namespace
 55      */
 56     var tabOverride = $.fn.tabOverride = function ( enable ) {
 57 
 58         // The jQuery object acts as an array of elements, so it can be passed
 59         // to TABOVERRIDE.enable() or TABOVERRIDE.disable().
 60         // If there are no arguments or enable is truthy, enable Tab Override,
 61         // otherwise, disable it.
 62         TABOVERRIDE.set(this, !arguments.length || enable);
 63 
 64         // Return the jQuery object
 65         return this;
 66     };
 67 
 68     /**
 69      * Gets or sets the tab size for all elements that have Tab Override enabled.
 70      * 0 represents the tab character.
 71      *
 72      * @param  {Number}          [size]  the tab size
 73      * @return {Number|Function}         the tab size or the tabOverride function
 74      *
 75      * @name tabSize
 76      * @function
 77      * @memberOf jQuery.fn.tabOverride
 78      */
 79     tabOverride.tabSize = TABOVERRIDE.tabSize;
 80 
 81     /**
 82      * Gets or sets the auto indent setting. True if each line should be
 83      * automatically indented (default = false).
 84      *
 85      * @param  {Boolean}          [enable]  whether auto indent should be enabled
 86      * @return {Boolean|Function}           whether auto indent is enabled or the
 87      *                                      tabOverride function
 88      *
 89      * @name autoIndent
 90      * @function
 91      * @memberOf jQuery.fn.tabOverride
 92      */
 93     tabOverride.autoIndent = TABOVERRIDE.autoIndent;
 94 }));
