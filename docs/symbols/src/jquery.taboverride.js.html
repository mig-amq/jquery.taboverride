 29 // Use AMD if available
 30 (function ( factory ) {
 31     "use strict";
 32 
 33     if ( typeof define === "function" && define.amd ) {
 34         // AMD - Register as a named module
 35         // Using a named module allows for non-AMD-aware concatenation
 36         define( "jquery.taboverride", [ "jquery", "taboverride" ], factory );
 37     } else {
 38         // No AMD - Use global variables instead
 39         factory( jQuery, TABOVERRIDE );
 40     }
 41 }(function ( $, TABOVERRIDE ) {
 42     "use strict";
 43 
 44     /**
 45      * the tabOverride method - Tabs will be overridden if enable is true.
 46      *
 47      * @param  {Boolean} [enable=true]  whether Tab Override should be enabled
 48      *                                  for the element(s)
 49      * @return {Object}                 the jQuery object
 50      *
 51      * @name tabOverride
 52      * @memberOf jQuery.fn
 53      * @function
 54      * @namespace
 55      */
 56     var tabOverride = $.fn.tabOverride = function ( enable ) {
 57 
 58         // Unbind the tabOverride event handlers so they are not bound
 59         // more than once
 60         var $textareas = this.filter("textarea").off(".tabOverride");
 61 
 62         // Only bind the tabOverride event handlers if the enable
 63         // argument is not specified or is truthy
 64         if ( !arguments.length || enable ) {
 65             $textareas
 66                 .on( "keydown.tabOverride", TABOVERRIDE.overrideKeyDown )
 67                 .on( "keypress.tabOverride", TABOVERRIDE.overrideKeyPress );
 68         }
 69 
 70         // Return the original jQuery object
 71         return this;
 72     };
 73 
 74     /**
 75      * Gets or sets the tab size for all elements that have Tab Override enabled.
 76      * 0 represents the tab character.
 77      *
 78      * @param  {Number}          [size]  the tab size
 79      * @return {Number|Function}         the tab size or the tabOverride function
 80      *
 81      * @name tabSize
 82      * @function
 83      * @memberOf jQuery.fn.tabOverride
 84      */
 85     tabOverride.tabSize = TABOVERRIDE.tabSize;
 86 
 87     /**
 88      * Returns the current tab size. 0 represents the tab character.
 89      *
 90      * @return {Number}  the size (length) of the tab string or 0 for the tab
 91      *                   character
 92      *
 93      * @name getTabSize
 94      * @function
 95      * @memberOf jQuery.fn.tabOverride
 96      * @deprecated since 2.0 - use tabSize() instead
 97      */
 98     tabOverride.getTabSize = function () {
 99         return this.tabSize();
100     };
101 
102     /**
103      * Sets the tab size for all elements that have Tab Override enabled.
104      * 0 represents the tab character. The initial value is 0.
105      *
106      * @param  {Number}   [size=0]  the tab size
107      * @return {Function}           the tabOverride function
108      *
109      * @name setTabSize
110      * @function
111      * @memberOf jQuery.fn.tabOverride
112      * @deprecated since 2.0 - use tabSize() instead
113      */
114     tabOverride.setTabSize = function ( size ) {
115         return this.tabSize( size || 0 );
116     };
117 
118     /**
119      * Gets or sets the auto indent setting. True if each line should be
120      * automatically indented (default = false).
121      *
122      * @param  {Boolean}          [enable]  whether auto indent should be enabled
123      * @return {Boolean|Function}           whether auto indent is enabled or the
124      *                                      tabOverride function
125      *
126      * @name autoIndent
127      * @function
128      * @memberOf jQuery.fn.tabOverride
129      */
130     tabOverride.autoIndent = TABOVERRIDE.autoIndent;
131 }));
